    # this.$lcs

    // packages/toast/index.js
import toastComponent from './src/main.vue';
let toast = {};
toast.install = function(Vue){
	let toastConsTructor = Vue.extend(toastComponent);
	Vue.prototype.$toast = function({message="",duration=2000,type="default"}){
		let toastDom = new toastConsTructor({
			el:document.createElement('div'),
			data(){
				return {
					message,
					isShow:true,
					type
				}
			}
		})
		let flag = document.getElementsByClassName('lcs-toast-wappper');
		if(!flag.length){
			let div = document.createElement('div');
			div.className = 'lcs-toast-wappper';
			document.body.appendChild(div)
		}
		document.getElementsByClassName('lcs-toast-wappper')[0].appendChild(toastDom.$el)
		setTimeout(()=>{
			toastDom.isShow = false
		},duration);
		setTimeout(()=>{
			document.getElementsByClassName('lcs-toast-wappper')[0].removeChild(toastDom.$el)
		},duration+250);
	}
}
export default toast;

    // toast/src/main.vue
<template>
	<div :class="[isShow?'fadein':'fadeout',typex]">
		{{message}}
	</div>
</template>

<script>
	export default {
		data() {
			return {
				message: 'default message',
				isShow: false,
				type: 'default'
			}
		},
		computed: {
			typex() {
				switch (this.type) {
					case 'success':
						return 'successType'
					case 'error':
						return 'errorType'
					default:
						return 'defaultType';
				}
			}
		}
	}
</script>

<style>

    # 导出组件库

    // packages/index.js
// 导入单个组件
import lcsTodo from './lcs-todo/index'

// 以数组的结构保存组件，便于遍历
const components = [
    lcsTodo
]

import toast from './toast/index'

// 定义 install 方法
const install = function (Vue) {
    if (install.installed) return
    install.installed = true
    // 遍历并注册全局组件
    components.map(component => {
        Vue.component(component.name, component)
    })
}

if (typeof window !== 'undefined' && window.Vue) {
    install(window.Vue)
}

export {
    // 导出的对象必须具备一个 install 方法
    install,
    lcsTodo,
	toast
}
