# Aes

对称加密，加密解密用一样的密钥

在egg中的aes

  1. 设置密钥
//config.default.js中设置aes 密钥变量，也可以放到别的地方
config.aes = {
   key:'key'
}
//这样我们就可以在controller里 let {ctx,app} = this;
//通过app.config.aes获取到该key值
  1. service/aes.js
const crypto = require('crypto')

// 加密
exports.aesEncrypt = (data, key) => {
  let cipher = crypto.createCipher('aes192', key)
  let crypted = cipher.update(data, 'utf8', 'hex')
  return cipher.final('hex')
}

// 解码
exports.aesDecrypt = (encrypt, key) => {
  let decipher = crypto.createDecipher('aes192', key);
  let decrypted = decipher.update(encrypt, 'hex', 'utf8');
  return decipher.final('utf8');
}
  1. controller/aes.js
const Controller = require('egg').Controller;

const {aesEncrypt,aesDecrypt} = require('../service/aes')

class AesController extends Controller {
    async getAes() {
        const { ctx,app } = this;
        const password = ctx.request.body.password;
        const aesed = aesEncrypt(password,app.config.aes.key)
        ctx.body = aesed;
    }

    async getAesDecrypt() {
        const { ctx,app } = this;
        const aes = ctx.request.body.aes;
        const xxDecrypt = aesDecrypt(aes,app.config.aes.key)
        ctx.body = xxDecrypt;
    }

}

module.exports = AesController;

TIP

导出 exports.aesEncrypt = (data, key) => {} exports.aesDecrypt = (encrypt, key) => {}

导入 const {aesEncrypt,aesDecrypt} = require('../service/aes')；

在egg中有约定了service，require service 再 class extends service，最后module.exports = RsaService;
