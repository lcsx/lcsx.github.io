    # koa-bodyparser 使request body能传输文件

    koa-bodyparser 只支持json/from-data格式，不支持文件，所有我们要替换他，要用koa-body

    npm i --save koa-body

    // app index.js
const koaBody = require('koa-body');
app.use(koaBody({
	multipart:true,  //开启支持文件
	formidable:{
		uploadDir:path.join(__dirname,'/public/uploads'),  //设置保存路径
		keepExtensions:true   //设置保留文件后缀名
	}
}))

    //controller中
class HomeCtl {
	index(ctx){
		ctx.body = '<h1>这是主页1</h1>'
	}
	upload(ctx){
		const file = ctx.request.files.file;    //这个file就是htnml中 对input设置的name的值
		ctx.body = "lcs";
	}
}

module.exports = new HomeCtl();

    TIP

    这里的file就是对应ctx.require.files.file 的file,也是postman中设置from-data key参数名为flie

    # koa-static 可以用http请求访问文件

    搭建静态服务，可通过http请求访问文件

    npm i --save koa-static

    // app index.js
const koaStatic = require('koa-static')
app.use(koaStatic(path.join(__dirname,'public')))
