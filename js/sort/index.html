    			let arrObjSortByPrporty = (arr,pramas=[],inverted=1)=>{
				
				const compare = (pramas) =>{
					
					const fn = (a,b)=>{	
						let n = 0;
						while(n<pramas.length){
							
							if(a[pramas[n]] - b[pramas[n]] !== 0){
								return (a[pramas[n]] - b[pramas[n]])*inverted
							}
							++n;
						}						
					}
					return fn;			
				}
				
				let result = arr.sort(compare(pramas))
