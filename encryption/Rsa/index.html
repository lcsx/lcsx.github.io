    # Rsa

    非对称加密 安全系数高

    # 生成私钥

    >openssl
>genrsa -out rsa_private_key.pem 1024

    # 根据私钥生成公钥

    >openssl
>rsa -in rsa_private_key.pem -pubout -out rsa_public_key.pem

    现在可以在当前目录看到两个密钥文件了

    # 在egg中使用

    1. 生成私钥，公钥（公钥类似于私钥的简化版）
    2. 引用公私钥的文件，用utf-8格式
    3. 加密/解密，用公钥加密，解密就要用私钥。用私钥加密，解密要用公钥。而且这4个操作各自要用的函数还不一样。

      操作(密钥,buffer)

    4. serviec层封装，controller调用

    3 加密解密

    const crypto = require('crypto');
const fs = require('fs');
const path = require('path');

//   privateEncrypt(密钥utf-8,buffer utf8)
//   crypto.privateEncrypt(密钥utf-8,buffer utf8).toString('hex')  转十六进制  也可toString('base64')
const Service = require('egg').Service;
class RsaService extends Service {
	async enc_by_prv(val){
		const privateKey = fs.readFileSync(path.resolve(__dirname, 'rsa_private_key.pem'), 'utf8');  //用utf8读取
		const buffer = Buffer.from(val, 'utf8');   //要加密的内容 buffer utf8
		const rsaval = crypto.privateEncrypt(privateKey, buffer).toString('hex');
		return rsaval;
	}
	async dec_by_pub(val){
		const publickKey = fs.readFileSync(path.resolve(__dirname, 'rsa_public_key.pem'), 'utf8');
		const buffer = Buffer.from(val, 'hex');
		const rsaval = crypto.publicDecrypt(publickKey, buffer).toString('utf8');
		return rsaval;
	}
}

module.exports = RsaService;

    TIP

    读取密钥用utf8 fs.readFileSync(path.resolve(__dirname, 'rsa_private_key.pem'), 'utf8');

    私钥加密 privateEncrypt，要加密的内容用buffer， 并用16进制返回 const buffer = Buffer.from(val, 'utf8'); crypto.privateEncrypt(privateKey, buffer).toString('hex');

    公钥解密 publicDecrypt ,要解密的密码上面是用hex十六进制的，所有解密时用的buffer也要用hex来读取 const buffer = Buffer.from(val, 'hex'); crypto.publicDecrypt(publickKey, buffer);

    公钥加密 crypto.publicEncrypt(pubKey, Buffer.from(message, 'utf8')).toString('hex');

    私钥解密 crypto.privateDecrypt(prvKey, enc_by_pub).toString('utf8');
