    # jwt的构成

    1. 头部 header
    2. 有效载荷 Payload
    3. 签名 signature

    形式 XXXXX.XXXXX.XXXX 为什么这么紧凑，因为用了base64编码国

    header包含

    1. typ:token的类型，这里固定为JWT
    2. alg:使用的hash算法，例如HMAC SHA256

    payload

    • 存储需要传递的信息，如用户ID、用户名等
    • 元数据，如国企时间，发布人等
    • 与header不同，payload可以加密

    signatuer

    • 对header 和 payload部分进行签名
    • 保证token在传输的过程种没有被篡改或者损坏

    # 手动认证与授权

    认证

    ...
const jsonwebtoken = require('jsonwebtoken');
const {key} = require('../config.js');

//auth 中间件
const auth = async (ctx,next)=>{
	const {authorization=""} = ctx.request.header;
	const token = authorization.replace("Bearer ","");
	try{
		const user = jsonwebtoken.verify(token,key);
		ctx.state.userInfo = user;  //ctx.state  约定俗成
	}catch(err){
		//TODO handle the exception
		ctx.throw(401,'token木有')
	}
	await next();
}

    controller中 jsonwebtoken的解密

    async login(ctx){
	ctx.verifyParams({
		name:{type:'string',required:true},
		password:{type:'string',required:true}
	});
	const user = await Users.findOne(ctx.request.body);
	if(!user) ctx.throw(401,'用户名或者密码不正确');
	const {name,_id} = user;
	const token = jsonwebtoken.sign({name,_id},key,{expiresIn:300});
	ctx.body = {token};
}
